Critical PurifiedNPC edge case (from PurifiedNPC.cs and EnemyStats.Purify()):
- `Purify()` fires `OnDeath?.Invoke()`, then sets `enabled = false`, then calls `PurifiedNPC.Activate()`.
- `PurifiedNPC.Activate()` does NOT call `Destroy(gameObject)`.
- Therefore `OnDestroy()` on any child UI script will NOT fire during Purify.
- Solution: `HandleDeath()` MUST call `Unbind()` before `SetActive(false)` so subscriptions are released synchronously during the `OnDeath` invocation.

Reference pattern from PlayerHUDBar.cs (Phase 1):
- Uses `GetComponentInParent<PlayerWaterStats>()` in Awake to auto-bind.
- Subscribes in Start(), unsubscribes in OnDestroy().
- For Phase 2 we switch to explicit Bind/Unbind because OnDestroy is unreliable on the Purify path.

    ```csharp
    using System.Collections;
    using UnityEngine;
    using UnityEngine.UI;

    /// <summary>
    /// World Space UI controller for enemy overhead bars.
    /// Owns HP bar, Corruption bar, and Sweet Spot overlay.
    /// Implements TECH-03 Bind/Unbind pattern to prevent event leaks on the Purify path
    /// (where PurifiedNPC.Activate() disables EnemyStats without destroying the GameObject,
    /// so OnDestroy on UI children is NOT guaranteed to fire).
    /// </summary>
    [RequireComponent(typeof(CanvasGroup))]
    public class EnemyWorldSpaceUI : MonoBehaviour
    {
        [Header("Canvas")]
        [SerializeField] private CanvasGroup canvasGroup;

        [Header("HP Bar")]
        [SerializeField] private Image hpFill;
        [SerializeField] private Color hpHealthyColor = new Color(0.25f, 0.72f, 0.35f, 1f); // #40B859
        [SerializeField] private Color hpLowColor     = new Color(0.85f, 0.22f, 0.22f, 1f); // #D93838

        [Header("Corruption Bar")]
        [SerializeField] private Image corruptionFill;

        [Header("Sweet Spot Overlay")]
        [SerializeField] private RectTransform sweetSpotOverlay;

        [Header("Fade")]
        [SerializeField] private float fadeInDuration = 0.15f;

        private EnemyStats _stats;
        private bool _revealed;
        private Coroutine _fadeCoroutine;

        private void Awake()
        {
            if (canvasGroup == null) canvasGroup = GetComponent<CanvasGroup>();

            var stats = GetComponentInParent<EnemyStats>();
            if (stats != null)
            {
                Bind(stats);
            }
            else
            {
                Debug.LogWarning($"[EnemyWorldSpaceUI] EnemyStats not found in parent of {gameObject.name}");
            }
        }

        /// <summary>Subscribe to EnemyStats events and initialize visuals. TECH-03.</summary>
        public void Bind(EnemyStats stats)
        {
            if (_stats != null) Unbind();

            _stats = stats;
            _stats.OnHpChanged         += HandleHpChanged;
            _stats.OnCorruptionChanged += HandleCorruptionChanged;
            _stats.OnDamaged           += HandleFirstDamage;
            _stats.OnDeath             += HandleDeath;

            // Initialize fills from current values
            HandleHpChanged(_stats.CurrentHp, _stats.MaxHp);
            HandleCorruptionChanged(_stats.CurrentCorruption, _stats.MaxCorruption);
            SetSweetSpotOverlay(_stats);

            // Start hidden until first hit (ENM-02, D-01)
            if (canvasGroup != null) canvasGroup.alpha = 0f;
            _revealed = false;
        }

        /// <summary>Unsubscribe from all EnemyStats events. Safe to call multiple times.</summary>
        public void Unbind()
        {
            if (_stats == null) return;
            _stats.OnHpChanged         -= HandleHpChanged;
            _stats.OnCorruptionChanged -= HandleCorruptionChanged;
            _stats.OnDamaged           -= HandleFirstDamage;
            _stats.OnDeath             -= HandleDeath;
            _stats = null;
        }

        private void OnDestroy() => Unbind();

        // ─── Event handlers ─────────────────────────────────────────────

        private void HandleHpChanged(float current, float max)
        {
            if (hpFill == null || max <= 0f) return;

            float ratio = Mathf.Clamp01(current / max);
            hpFill.fillAmount = ratio;

            // Lerp color from healthy to low when ratio drops below 0.5
            if (ratio >= 0.5f)
            {
                hpFill.color = hpHealthyColor;
            }
            else
            {
                float t = 1f - (ratio * 2f); // ratio 0.5 → t=0 (healthy); ratio 0 → t=1 (low)
                hpFill.color = Color.Lerp(hpHealthyColor, hpLowColor, t);
            }
        }

        private void HandleCorruptionChanged(float current, float max)
        {
            if (corruptionFill == null || max <= 0f) return;
            corruptionFill.fillAmount = Mathf.Clamp01(current / max);
        }

        private void HandleFirstDamage()
        {
            if (_revealed) return;
            _revealed = true;

            if (canvasGroup == null) return;
            if (_fadeCoroutine != null) StopCoroutine(_fadeCoroutine);
            _fadeCoroutine = StartCoroutine(FadeInCanvas());
        }

        private void HandleDeath()
        {
            // CRITICAL: Unbind BEFORE SetActive(false) so the Purify path
            // (where EnemyStats.enabled = false and no Destroy follows)
            // does not leave dangling event subscriptions.
            Unbind();

            if (_fadeCoroutine != null)
            {
                StopCoroutine(_fadeCoroutine);
                _fadeCoroutine = null;
            }

            gameObject.SetActive(false);
        }

        // ─── Sweet Spot overlay ─────────────────────────────────────────

        private void SetSweetSpotOverlay(EnemyStats stats)
        {
            if (sweetSpotOverlay == null) return;

            float effectiveMin = Mathf.Clamp01(stats.basePurificationMin - stats.bonusPurificationMargin);
            float effectiveMax = Mathf.Clamp01(stats.basePurificationMax + stats.bonusPurificationMargin);

            sweetSpotOverlay.anchorMin = new Vector2(effectiveMin, 0f);
            sweetSpotOverlay.anchorMax = new Vector2(effectiveMax, 1f);
            // CRITICAL: zero the offsets so retained pixel padding does not shift the overlay
            sweetSpotOverlay.offsetMin = Vector2.zero;
            sweetSpotOverlay.offsetMax = Vector2.zero;
        }

        // ─── Fade-in coroutine ──────────────────────────────────────────

        private IEnumerator FadeInCanvas()
        {
            float elapsed = 0f;
            if (canvasGroup != null) canvasGroup.alpha = 0f;

            while (elapsed < fadeInDuration)
            {
                elapsed += Time.deltaTime;
                float t = Mathf.Clamp01(elapsed / fadeInDuration);
                if (canvasGroup != null) canvasGroup.alpha = t;
                yield return null;
            }

            if (canvasGroup != null) canvasGroup.alpha = 1f;
            _fadeCoroutine = null;
        }
    }
    ```
